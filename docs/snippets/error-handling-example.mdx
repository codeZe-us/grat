```typescript
import { SubmissionFailedError, RateLimitError, FrozenEntryError } from '@grat-official-sdk/sdk';

try {
  const result = await grat.sponsor(tx);
} catch (error) {
  if (error instanceof SubmissionFailedError) {
    // Transaction rejected by Stellar network
    console.error("Stellar Error:", error.message);
    console.debug("Result Codes:", error.details);
  } else if (error instanceof RateLimitError) {
    // Relay rate limit exceeded
    console.log(`Retrying in ${error.retryAfter} seconds...`);
  } else if (error instanceof FrozenEntryError) {
    // Network-level block on this ledger entry (Protocol 26)
    console.warn("Action restricted: one or more ledger entries are frozen.");
  }
}
```
