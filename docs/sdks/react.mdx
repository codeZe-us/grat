## Using the TypeScript SDK in React Today

The `@grat-official-sdk/sdk` package is framework-agnostic. You can use it directly in any React component. Here is a complete example:

```tsx
import { useState } from 'react';
import { Grat } from '@grat-official-sdk/sdk';
import { Transaction } from '@stellar/stellar-sdk';

const grat = new Grat({ relayUrl: 'https://your-relay.example.com' });

export function SendButton({ signedXdr }: { signedXdr: string }) {
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState<string | null>(null);
  const [hash, setHash] = useState<string | null>(null);

  const handleSend = async () => {
    setLoading(true);
    setError(null);
    try {
      const tx = new Transaction(signedXdr, 'Test SDF Network ; September 2015');
      const result = await grat.sponsor(tx);
      setHash(result.hash);
    } catch (err) {
      setError(err instanceof Error ? err.message : 'Transaction failed');
    } finally {
      setLoading(false);
    }
  };

  return (
    <div>
      <button onClick={handleSend} disabled={loading}>
        {loading ? 'Sending...' : 'Send'}
      </button>
      {hash && <p>Confirmed: {hash}</p>}
      {error && <p>Error: {error}</p>}
    </div>
  );
}
```

Install the SDK:

```bash
npm install @grat-official-sdk/sdk @stellar/stellar-sdk
```

That is everything you need to sponsor transactions from React today.

## React SDK (Phase 4)

The hooks-based React SDK will ship in Phase 4. The planned API is documented below. This is not available yet.

### Planned Hooks

#### `useSponsor`

```tsx
const { sponsor, loading, error, result } = useSponsor();
```

Wraps `grat.sponsor()` with React state management. Handles loading and error states automatically.

```tsx
function SendButton({ tx }) {
  const { sponsor, loading, error, result } = useSponsor();

  return (
    <button onClick={() => sponsor(tx)} disabled={loading}>
      {loading ? 'Sending...' : 'Send'}
      {result && <span>Hash: {result.hash}</span>}
      {error && <span>Error: {error.message}</span>}
    </button>
  );
}
```

#### `useRelayStatus`

```tsx
const { status, loading } = useRelayStatus();
```

Polls the `/health` endpoint and returns the current relay status.

```tsx
function RelayBadge() {
  const { status } = useRelayStatus();
  return <span>{status?.status === 'ok' ? 'Online' : 'Offline'}</span>;
}
```

#### `useCreditBalance`

```tsx
const { balance, refresh } = useCreditBalance();
```

Returns the current credit balance for the authenticated API key and a `refresh` function to re-fetch it.

### Planned Provider

All hooks will require a `GratProvider` at the root of your app:

```tsx
import { GratProvider } from '@grat-official-sdk/react';

function App() {
  return (
    <GratProvider relayUrl="https://your-relay.example.com" apiKey="grat_your_key">
      <YourApp />
    </GratProvider>
  );
}
```
